        \textbf{\textit{Diffential Fomulas:}}
        \begin{itemize}
            \item $ \frac{d}{dx}(c) = 0$
            \item $ \frac{d}{dx}(x) = 1$
            \item $ \frac{d}{dx}(x^n) = n \cdot x^{n-1}$
            \item $ \frac{d}{dx}[c \cdot f(x)] = c \cdot \frac{d}{dx}[f(x)]$
            \item $ \frac{d}{dx}[f(x) \pm g(x)] = \frac{d}{dx}f(x)\pm \frac{d}{dx}g(x)$
        \end{itemize}
